\begin{frame}[t]{Why post-train scientific agents?}
  \vspace{2mm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.47\textwidth}
      {\Large\color{DeckBlue}\textbf{Agent harness}}
      \vspace{3mm}
      \begin{itemize}
        \item Defines tools and environments
        \item Manages context and collaboration
        \item Determines how capability is executed
      \end{itemize}
    \end{column}
    \begin{column}{0.06\textwidth}
      \vspace{12mm}
      \centering{\Huge\color{DeckGold}$+$}
    \end{column}
    \begin{column}{0.47\textwidth}
      {\Large\color{DeckTeal}\textbf{Post-training}}
      \vspace{3mm}
      \begin{itemize}
        \item Changes the model's behavior
        \item Adds domain-specific intelligence
        \item Determines what can be learned and retained
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{3mm}
  \begin{center}
    {\LARGE\textbf{AI scientist} $=$ execution system $+$ adaptive model intelligence}
  \end{center}

  \takeaway{A better harness helps a model reach its capability; post-training changes the capability itself.}
\end{frame}
