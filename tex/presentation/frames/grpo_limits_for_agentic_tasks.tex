\begin{frame}[t]{Why pure GRPO becomes inefficient for agents}
  \vspace{2mm}
  \begin{enumerate}
    \setlength{\itemsep}{4mm}
    \item \textbf{Low diversity.} If sibling attempts use the same strategy and receive the same reward, every relative advantage is near zero.
    \item \textbf{Sparse information.} A costly rollout returns a number, not a diagnosis or corrected behavior.
    \item \textbf{Expensive comparison.} Each prompt needs several complete long rollouts before one update can be formed.
  \end{enumerate}

  \vspace{5mm}
  \begin{beamercolorbox}[wd=\textwidth,sep=2mm]{takeaway}
    \large GRPO is strongest when outcomes are cheap to verify and the policy already produces meaningfully different attempts.
  \end{beamercolorbox}

  \slidesource{DeepSeek-R1 demonstrates the strong case; later analyses identify group diversity and credit-density limits \citep{guo2025deepseekr1,shen2026firstprinciplesgrpo}.}
\end{frame}
