\begin{frame}[t]{SFT: learn by imitating a demonstration}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      {\large\textbf{Algorithm}}
      \begin{enumerate}
        \item Collect a desired agent trace.
        \item Use user and tool messages as context.
        \item Train on demonstrated assistant tokens only.
      \end{enumerate}

      \vspace{3mm}
      \[
        \mathcal L_{\mathrm{SFT}}
        =-\sum_t m_t\log \pi_\theta(y_t\mid y_{<t},x)
      \]
    \end{column}
    \begin{column}{0.44\textwidth}
      {\large\textbf{Agent example}}
      \vspace{2mm}
      \begin{tikzpicture}[node distance=3mm]
        \node[flowbox,text width=42mm] (user) {User task};
        \node[flowbox,text width=42mm,below=of user] (tool) {Tool result\\\scriptsize visible, but masked};
        \node[signalbox,text width=42mm,below=of tool] (action) {Desired model action\\\scriptsize trained target};
        \draw[flowarrow] (user) -- (tool);
        \draw[flowarrow] (tool) -- (action);
      \end{tikzpicture}
    \end{column}
  \end{columns}

  \vspace{1mm}
  \begin{beamercolorbox}[wd=\textwidth,sep=1.2mm]{takeaway}
    \small
    \textcolor{DeckRed}{\textbf{Coverage:}} demonstrations show the desired path, not recovery from self-generated errors.\\[0.7mm]
    \textcolor{DeckRed}{\textbf{Catastrophic forgetting:}} forcing every target token can overwrite prior behavior. RL and OPD usually shift less because they train on the model's own samples.
  \end{beamercolorbox}
\end{frame}
