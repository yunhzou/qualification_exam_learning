\begin{frame}[t]{Conclusion: long-horizon agents need denser feedback}
  \vspace{1mm}
  {\Large\textbf{A terminal score verifies the outcome; it rarely explains how to improve.}}

  \vspace{5mm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.30\textwidth}
      {\large\color{DeckRed}\textbf{Outcome reward}}
      \vspace{2mm}

      Confirms success or failure, but compresses an entire trajectory into one number.
    \end{column}
    \begin{column}{0.32\textwidth}
      {\large\color{DeckGold}\textbf{Process feedback}}
      \vspace{2mm}

      Assigns credit at the trajectory, trace, or action level so good decisions survive later mistakes.
    \end{column}
    \begin{column}{0.32\textwidth}
      {\large\color{DeckTeal}\textbf{Textual feedback}}
      \vspace{2mm}

      Carries diagnosis, rationale, and corrective strategy that can directly reshape later behavior.
    \end{column}
  \end{columns}

  \vspace{6mm}
  \begin{beamercolorbox}[wd=\textwidth,sep=2mm]{takeaway}
    \small \textbf{My position:} combine verifiable outcomes with process-level and textual feedback, then automate the loop for continual adaptation. Humans learn from explanations, hints, and targeted practice---not only from ``right'' or ``wrong'' labels.
  \end{beamercolorbox}
\end{frame}
