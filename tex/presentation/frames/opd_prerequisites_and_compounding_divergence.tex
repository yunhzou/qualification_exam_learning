\begin{frame}[t]{OPD prerequisite: student states must remain teachable}
  \vspace{0mm}
  {\centering
  \begin{tikzpicture}[node distance=5mm]
    \node[flowbox,text width=24mm] (task) {Task state\\\scriptsize required data present};
    \node[warnbox,text width=27mm,right=of task] (delete) {Student calls\\\texttt{delete(data/)}};
    \node[warnbox,text width=25mm,right=of delete] (gone) {Environment\\\scriptsize data is gone};
    \node[signalbox,text width=30mm,right=of gone] (recover) {Teacher logits\\\scriptsize recovery from damage};
    \draw[flowarrow] (task) -- (delete);
    \draw[flowarrow] (delete) -- (gone);
    \draw[flowarrow] (gone) -- (recover);
  \end{tikzpicture}
  \par}

  \vspace{1mm}
  \begin{beamercolorbox}[wd=\textwidth,sep=1.2mm]{takeaway}
    \centering\small The teacher can reject the destructive call at that token, but after execution it can only score continuations from the damaged state.
  \end{beamercolorbox}

  \vspace{0mm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.47\textwidth}
      {\large\color{DeckTeal}\textbf{Meaningful supervision}}
      \small
      \begin{itemize}
        \item At the bad action, teacher logits favor a safe alternative.
        \item From a recoverable mistake, the teacher can teach correction behavior.
      \end{itemize}
    \end{column}
    \begin{column}{0.49\textwidth}
      {\large\color{DeckRed}\textbf{When it weakens}}
      \small
      \begin{itemize}
        \item A weak student visits prefixes far from the teacher's useful solution manifold.
        \item Later-token guidance becomes damage control rather than the intended strategy.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{1mm}
  {\footnotesize\textbf{Need:} policy overlap, teacher competence on visited prefixes, recoverable segments, and iterative resampling.}
\end{frame}
