\begin{frame}[t]{Chess exposes trajectory-level credit failure}
  \vspace{1mm}
  {\centering
  \begin{tikzpicture}[node distance=5mm]
    \node[flowbox,text width=22mm] (state) {Position $s_t$};
    \node[signalbox,text width=28mm,right=of state] (good) {Good move\\\scriptsize wins a rook};
    \node[warnbox,text width=28mm,right=of good] (bad) {Eight moves later\\\scriptsize hangs the queen};
    \node[warnbox,text width=21mm,right=of bad] (loss) {Final result\\\textbf{loss}};
    \draw[flowarrow] (state) -- (good);
    \draw[flowarrow] (good) -- (bad);
    \draw[flowarrow] (bad) -- (loss);

    \node[draw=DeckRed,fill=DeckRed!7,rounded corners=2pt,align=center,
      text width=95mm,minimum height=9mm,font=\small,below=7mm of good,
      xshift=23mm] (grpo)
      {\textbf{GRPO:} the lost game gives one negative trajectory advantage $A_i<0$ to \emph{both} moves.};
    \draw[-{Latex[length=2mm]},DeckRed,line width=0.9pt] (grpo.north west) to[bend left=16] (good.south);
    \draw[-{Latex[length=2mm]},DeckRed,line width=0.9pt] (grpo.north east) to[bend right=16] (bad.south);
  \end{tikzpicture}
  \par}

  \vspace{4mm}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.31\textwidth}
      {\large\color{DeckRed}\textbf{1. Outcome: $G_t$}}
      \vspace{1mm}

      \small $G_t$ sums future rewards. With only terminal chess reward, every move inherits who won; a later blunder punishes the earlier good move.
    \end{column}
    \begin{column}{0.32\textwidth}
      {\large\color{DeckGold}\textbf{2. State-relative}}
      \vspace{1mm}

      \small Expected win $+$ actual win gives small $G_t-V(s_t)$; an unlikely win gives a positive surprise. It still credits the realized future as a whole.
    \end{column}
    \begin{column}{0.32\textwidth}
      {\large\color{DeckTeal}\textbf{3. Stepwise critic}}
      \vspace{1mm}

      \small $r_t+\gamma V(s_{t+1})-V(s_t)$ asks whether this move improved expected return, but trusts the critic.
    \end{column}
  \end{columns}
\end{frame}
