\begin{frame}[t]{Agentic RL: compaction creates multiple traces}
  \begin{center}
    \begin{tikzpicture}[node distance=4mm and 6mm]
      \node[flowbox,text width=33mm] (call1) {Model call 1\\\scriptsize screenshots 1--2};
      \node[warnbox,text width=28mm,right=of call1] (compact) {Context compaction};
      \node[flowbox,text width=36mm,right=of compact] (call2) {Model call 2\\\scriptsize summary + screenshots 3--4};
      \draw[flowarrow] (call1) -- (compact);
      \draw[flowarrow] (compact) -- (call2);

      \node[font=\small,text=DeckRed,below=1mm of compact] (warning)
        {Context changed: start a new trace};
      \node[signalbox,text width=35mm,below=6mm of call1] (trace1)
        {Valid trace 1\\\scriptsize exact first context};
      \node[signalbox,text width=35mm,below=6mm of call2] (trace2)
        {Valid trace 2\\\scriptsize exact compacted context};
    \end{tikzpicture}
  \end{center}

  \begin{itemize}
    \item Polar records each model call with the exact context and behavior-policy probabilities that produced it \citep{xu2026polar}.
    \item Trace splitting produces variable counts and lengths; GLM-5.2 uses critic-based PPO with token-level advantages and loss for this ragged setting \citep{glm52team2026}.
  \end{itemize}
\end{frame}
