\begin{figure}[p]
  \centering
  \resizebox{!}{0.72\textheight}{%
  \begin{tikzpicture}[
      flow/.style={-Latex, line width=0.8pt, draw=black!55},
      retry/.style={-Latex, line width=1pt, draw=orange!85},
      learn/.style={-Latex, line width=1pt, draw=green!60!black},
      fail/.style={-Latex, line width=1pt, draw=red!75},
      humanflow/.style={-Latex, line width=1pt, draw=violet!60},
      box/.style={draw=black!55, rounded corners=2pt, line width=0.8pt,
                  fill=white, align=center, text width=4.4cm,
                  minimum height=1.05cm, font=\small, inner sep=3pt},
      smallbox/.style={box, text width=0.55cm, minimum height=0.65cm,
                      font=\small, inner sep=1pt},
      note/.style={font=\scriptsize, align=center, fill=blue!8, inner sep=1.5pt}
    ]
    % The layout deliberately follows the original concept figure: one main learning loop.
    \filldraw[fill=black!7, draw=black!55, line width=0.9pt, rounded corners=3pt]
      (-5.2,0.25) rectangle (5.2,-18.0);
    \filldraw[fill=blue!10, draw=cyan!70!blue, line width=1pt, rounded corners=3pt]
      (-4.55,-3.15) rectangle (4.55,-17.45);

    \node[font=\Large, anchor=south east] at (4.75,0.35) {Agent};
    \node[font=\large, anchor=north east] at (4.2,-3.25) {Adaptation};

    \node[box, fill=violet!12, draw=violet!55, text width=4.6cm,
          minimum height=1.35cm] (objective) at (0,1.75)
      {\textbf{Objective}\\externally defined\\[-1pt]
       \scriptsize e.g., by a human scientist};
    \draw[humanflow] (objective.south) -- (0,0.3);

    \node[box] (create) at (0,-0.75) {\textbf{Create Tasks}};
    \node[smallbox] (taskone) at (-2.0,-2.15) {$t_1$};
    \node[font=\large] at (0,-2.15) {$\cdots$};
    \node[smallbox] (taskn) at (2.0,-2.15) {$t_n$};
    \coordinate (split) at (0,-1.55);
    \coordinate (merge) at (0,-2.85);
    \draw[flow] (create.south) -- (split);
    \draw[flow] (split) -| (taskone.north);
    \draw[flow] (split) -| (taskn.north);
    \draw[flow] (taskone.south) |- (merge);
    \draw[flow] (taskn.south) |- (merge);

    \node[box, fill=blue!4] (receive) at (0,-4.1) {\textbf{Receive Task}};
    \node[box, fill=blue!4] (attempt) at (0,-6.15)
      {\textbf{Attempt Task}\\[-1pt]
       \scriptsize retrieve relevant knowledge};
    \node[box, fill=blue!4] (reflect) at (0,-8.65)
      {\textbf{Reflection}\\[-1pt]
       \scriptsize identify mistake or gap};
    \node[box, fill=blue!4] (curriculum) at (0,-11.25)
      {\textbf{Build Curriculum}\\[-1pt]
       \scriptsize content, practice tasks, and evaluations};
    \node[box, fill=green!7, draw=green!50!black] (posttrain) at (0,-13.65)
      {\textbf{Continual Post-Training}\\[-1pt]
       \scriptsize update model parameters};
    \node[box, fill=blue!4] (practice) at (0,-16.0)
      {\textbf{Practice Tasks}\\[-1pt]
       \scriptsize benchmarks};

    \draw[flow] (merge) -- (receive.north);
    \draw[flow] (receive) -- (attempt);
    \draw[fail] (attempt) -- node[note, text=red!75] {failure} (reflect);
    \draw[learn] (reflect) -- node[note, text=green!55!black]
      {capability gap} (curriculum);
    \draw[flow] (curriculum) -- (posttrain);
    \draw[flow] (posttrain) -- (practice);

    \draw[retry] (reflect.west) -- ++(-1.25,0) |- node[note, text=orange!85,
      pos=0.65] {local mistake\\retry} (attempt.west);
    \draw[fail] (practice.west) -- ++(-1.75,0) |- node[note, text=red!75,
      pos=0.36] {failure\\recursive learning} (reflect.west);
    \draw[learn] (practice.east) -- ++(1.65,0) |- node[note, text=green!55!black,
      pos=0.34] {success\\retry original task} (attempt.east);

    \node[box, fill=violet!10, draw=violet!55, text width=3.0cm,
          minimum height=1.25cm, font=\scriptsize] (human) at (7.0,-8.65)
      {Human discretionary\\interrupt or injection};
    \draw[humanflow] (human.west) -- (reflect.east);
  \end{tikzpicture}%
  }
  \caption{Automated continual post-training loop, rebuilt from the initial concept
  figure. The agent diagnoses failures, constructs targeted learning material, updates
  the model through continual post-training, and evaluates the resulting capability
  before retrying the original task.}
  \label{fig:auto-post-training-loop}
\end{figure}
